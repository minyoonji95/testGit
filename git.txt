git init 
git add git.txt // не добавляет нихуя, но зачемто нужно писать комманду после добавления файла вручную
 git commit -m 'add git.txt' // коммитим изменения, чтобы было видно что мы делаем

 git remote show origin // пока непонятная команда, но нужна чтобы все четко синхронизировалось на гитхабе, пишется перед тем как запушить в гитхаб
 git remote set-url origin https://github.com/minyoonji95/testGit.git // если сбились настройки в вс коде, то эта команда помогает определить корневой католог на гитхабе

 git push -u origin main // пушатся изменения на гитхабе
 git clone git@github.com:<ИМЯ НА ГИТХАБЕ>/git.git // склонировать изменения с публичного репозитория в локальный репозиторий
 git pull --rebase // если появились изменения в публичном репозитории, то эта команда поможет обновить их в локальном репозитории

 git status // показывает текущий статус
